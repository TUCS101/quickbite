g.onload=function(){ const transcriptEl=document.getElementById("transcriptText"); const vapi = window.vapiSDK.run({ apiKey, assistant, container:"#vapi-container" }); // Listen for messages from Vapi vapi.on("message",(msg)=>{ // User speaking if(msg.type==="transcript" && msg.transcriptType==="partial"){ transcriptEl.innerText=msg.text; } // Finalized sentences if(msg.type==="transcript" && msg.transcriptType==="final"){ transcriptEl.innerText += "\n" + msg.text; } }); };
